program binaryProgram;

var
    int arr[5], iNum, iN, indexResult; 

module int binarySearch(int iL, int iR, int iX) 
var int mid;
{ 
    if (iR >= iL) then{ 

        mid = (iL + iR) / 2; 
        if (arr[mid] == iX) then {
            return(mid);
        } 

        if (arr[mid] > iX) then{
            return(binarySearch(iL, mid - 1, iX));

        } else {
            return(binarySearch(mid + 1, iR, iX));
        }

    } 
    return(-1);
} 
  
main(){ 
    arr[0] = 2; 
    arr[1] = 3; 
    arr[2] = 4; 
    arr[3] = 10; 
    arr[4] = 40; 
    
    iNum = 10; 
    iN = 5; 

    indexResult = binarySearch(0, iN - 1, iNum);

    if(indexResult == -1) then {
        write("The element does not exist within the array");
    } else{
        write("Element is at the index", indexResult);
    }
}
